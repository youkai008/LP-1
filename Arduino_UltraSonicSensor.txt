
const int trigPin = 9;
const int LED = 11;
const int echoPin = 10;
long duration;
int distance;

void setup() 
{
    Serial.println("Started");
    pinMode(trigPin, OUTPUT); 
    pinMode(echoPin, INPUT); 
    pinMode(LED, OUTPUT);
    Serial.begin(4800); // Starts the serial communication
}

void loop() 
{
    // Clears the trigPin
    digitalWrite(trigPin, LOW);
    delayMicroseconds(2);

    // Sets the trigPin on HIGH state for 10 micro seconds
    digitalWrite(trigPin, HIGH);
    delayMicroseconds(10);
    digitalWrite(trigPin, LOW); // Again bringing the trigpin to low state..

    // Reads the echoPin, returns the sound wave travel time in microseconds
    duration = pulseIn(echoPin, HIGH);
    distance = duration * 0.034 / 2;
    Serial.print("Distance: ");    
    Serial.println(distance);

    if(distance<100 && distance>20)
    {
        // When distance is intermediate, the blink rate will be low.
        digitalWrite(LED, HIGH);
        delay(200);
        digitalWrite(LED, LOW);
        delay(200);
    }
    if(distance<=20)
    {
        // Now as the object comes much closer the blink rate increases.
        digitalWrite(LED, HIGH);
        delay(50);
        digitalWrite(LED, LOW);
        delay(50);
    }
}
